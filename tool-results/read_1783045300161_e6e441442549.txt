    51→/*  Scroll-reveal hook                                                 */
    52→/* ------------------------------------------------------------------ */
    53→function useReveal() {
    54→  const ref = useRef<HTMLDivElement>(null);
    55→
    56→  useEffect(() => {
    57→    const el = ref.current;
    58→    if (!el) return;
    59→    const children = el.querySelectorAll(".reveal");
    60→    const observer = new IntersectionObserver(
    61→      (entries) => {
    62→        entries.forEach((e) => {
    63→          if (e.isIntersecting) {
    64→            e.target.classList.add("visible");
    65→            observer.unobserve(e.target);
    66→          }
    67→        });
    68→      },
    69→      { threshold: 0.15 }
    70→    );
    71→    children.forEach((c) => observer.observe(c));
    72→    return () => observer.disconnect();
    73→  }, []);
    74→
    75→  return ref;
    76→}

    80→/* ------------------------------------------------------------------ */
    81→function Counter({ end, suffix = "" }: { end: number; suffix?: string }) {
    82→  const [count, setCount] = useState(0);
    83→  const ref = useRef<HTMLSpanElement>(null);
    84→  const started = useRef(false);
    85→
    86→  useEffect(() => {
    87→    const el = ref.current;
    88→    if (!el) return;
    89→    const observer = new IntersectionObserver(
    90→      ([entry]) => {
    91→        if (entry.isIntersecting && !started.current) {
    92→          started.current = true;
    93→          const duration = 2000;
    94→          const startTime = performance.now();
    95→          const step = (now: number) => {
    96→            const elapsed = now - startTime;
    97→            const progress = Math.min(elapsed / duration, 1);
    98→            const eased = 1 - Math.pow(1 - progress, 3);
    99→            setCount(Math.round(eased * end));
   100→            if (progress < 1) requestAnimationFrame(step);
   101→          };
   102→          requestAnimationFrame(step);
   103→        }
   104→      },
   105→      { threshold: 0.5 }
   106→    );
   107→    observer.observe(el);
   108→    return () => observer.disconnect();
   109→  }, [end]);
   110→
   111→  return (
   112→    <span ref={ref}>
   113→      {count}
   114→      {suffix}
   115→    </span>
   116→  );
   117→}
